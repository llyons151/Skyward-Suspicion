{

};

void AirshipAnimation::displayIdleAnimation()
{
    for(const std::string& szFrame : szAirshipFrames)
    {
        std::cout << "\033[2J\033[1;1H" << szFrame << std::flush;
        std::this_thread::sleep_for(std::chrono::milliseconds(150));
    };
};
